       ```javascript
           }
         }

       });
       ```

    2. Create backend/src/services/transactionRecorder.js:
       ```javascript
       const Transaction = require('../models/Transaction');
       
       /**
        * Create a pending transaction record
        * @param {Object} params
        * @param {string} params.deviceId - Device identifier
        * @param {string} params.userId - User ObjectId
        * @param {string} params.walletAddress - User's connected wallet
        * @param {string} params.action - 'on' or 'off'
        * @returns {Promise<Transaction>}
        */
       async function createTransaction({ deviceId, userId, walletAddress, action }) {
         const transaction = new Transaction({
           deviceId,
           userId,
           walletAddress: walletAddress.toLowerCase(),
           action,
           timestamp: new Date(),
           status: 'pending'
         });
         
         await transaction.save();
         console.log(`[TXN] Created pending transaction: ${transaction._id}`);
         
         return transaction;
       }
       
       /**
        * Confirm a transaction with blockchain hash
        * @param {string} txId - MongoDB transaction ID
        * @param {Object} confirmation
        * @param {string} confirmation.txHash - Blockchain transaction hash
        * @param {number} confirmation.blockNumber - On-chain block number
        */
       async function confirmTransaction(txId, { txHash, blockNumber }) {
         const transaction = await Transaction.findByIdAndUpdate(
           txId,
           {
             txHash,
             blockNumber,
             status: 'confirmed'
           },
           { new: true }
         );
         
         if (transaction) {
           console.log(`[TXN] Confirmed transaction: ${txId} -> ${txHash}`);
         }
         
         return transaction;
       }
       
       /**
        * Mark a transaction as failed
        * @param {string} txId - MongoDB transaction ID
        * @param {string} error - Error message
        */
       async function failTransaction(txId, error) {
         const transaction = await Transaction.findByIdAndUpdate(
           txId,
           {
             status: 'failed'
           },
           { new: true }
         );
         
         console.error(`[TXN] Failed transaction: ${txId} - ${error}`);
         return transaction;
       }
       
       /**
        * Get transaction history for a device
        * @param {string} deviceId
        * @param {Object} options
        * @param {number} options.limit - Results per page (default 50)
        * @param {number} options.skip - Offset (default 0)
        * @param {string} options.sort - 'asc' or 'desc' (default 'desc')
        */
       async function getTransactionHistory(deviceId, options = {}) {
         const { limit = 50, skip = 0, sort = 'desc' } = options;
         const sortOrder = sort === 'asc' ? 1 : -1;
         
         const [transactions, total] = await Promise.all([
           Transaction.find({ deviceId })
             .sort({ timestamp: sortOrder })
             .skip(skip)
             .limit(limit)
             .lean(),
           Transaction.countDocuments({ deviceId })
         ]);
         
         return {
           transactions,
           total,
           limit,
           skip,
           hasMore: skip + limit < total
         };
       }
       
       /**
        * Get transaction by blockchain hash
        * @param {string} txHash
        */
       async function getTransactionByHash(txHash) {
         return Transaction.findOne({ txHash }).lean();
       }
       
       /**
        * Get transaction summary for a device
        * @param {string} deviceId
        */
       async function getTransactionSummary(deviceId) {
         const stats = await Transaction.aggregate([
           { $match: { deviceId, status: 'confirmed' } },
           { $group: {
             _id: null,
             totalTransactions: { $sum: 1 },
             totalOnDuration: { $sum: { $cond: [{ $eq: ['$action', 'on'] }, '$duration', 0] } },
             avgOnDuration: { $avg: { $cond: [{ $eq: ['$action', 'on'] }, '$duration', null] } },
             confirmedCount: { $sum: 1 },
             pendingCount: { $sum: { $cond: [{ $eq: ['$status', 'pending'] }, 1, 0] } },
             lastAction: { $last: '$action' },
             lastTimestamp: { $last: '$timestamp' }
           }}
         ]);
         
         if (stats.length === 0) {
           return {
             totalTransactions: 0,
             totalOnDuration: 0,
             avgOnDuration: 0,
             confirmedCount: 0,
             pendingCount: 0,
             lastAction: null,
             lastTimestamp: null
           };
         }
         
         return {
           ...stats[0],
           lastTimestamp: stats[0].lastTimestamp?.toISOString()
         };
       }
       
       module.exports = {
         createTransaction,
         confirmTransaction,
         failTransaction,
         getTransactionHistory,
         getTransactionByHash,
         getTransactionSummary
       };
       ```

       ```javascript
       const { ethers } = require('ethers');
       const { confirmTransaction, failTransaction, getTransactionByHash } = require('../services/transactionRecorder');
       
       const CONTRACT_ABI = [
         'event DeviceAction(bytes32 indexed deviceId, string action, uint256 timestamp, address indexed user, bytes32 indexed dbRecordId)'
       ];
       
       let provider = null;
       let contract = null;
       let listener = null;
       
       /**
        * Initialize blockchain listener for transaction confirmations
        * @param {string} rpcUrl - Infura Sepolia RPC URL
        * @param {string} contractAddress - Deployed IOTEventLog address
        */
       function initBlockchainListener(rpcUrl, contractAddress) {
         if (!rpcUrl || !contractAddress) {
           console.warn('[BLOCKCHAIN] RPC URL or contract address not configured');
           return;
         }
         
         provider = new ethers.JsonRpcProvider(rpcUrl);
         contract = new ethers.Contract(contractAddress, CONTRACT_ABI, provider);
         
         console.log(`[BLOCKCHAIN] Listening for events on ${contractAddress}`);
       }
       
       /**
        * Start listening for DeviceAction events
        * Call this after server starts to track confirmations
        */
       function startListening() {
         if (!contract) {
           console.warn('[BLOCKCHAIN] Listener not initialized');
           return;
         }
         
         // Listen for new events (past events would need historical query)
         contract.on('DeviceAction', async (deviceId, action, timestamp, user, dbRecordId, tx) => {
           try {
             console.log(`[BLOCKCHAIN] Event received: ${action} for device ${deviceId}`);
             
             // Find the pending transaction by MongoDB ID (stored as dbRecordId)
             const txHash = tx.transactionHash;
             const blockNumber = tx.blockNumber;
             
             // MongoDB ObjectId is stored as bytes32 in the event
             // We need to match by txHash since direct ID lookup may not work
             const pendingTx = await getTransactionByHash(null); // Will need different approach
             
             // Alternative: Poll pending transactions for confirmation
             // This is handled by the polling approach in startPollingPending
           } catch (error) {
             console.error('[BLOCKCHAIN] Error processing event:', error);
           }
         });
         
         console.log('[BLOCKCHAIN] Event listener started');
       }
       
       /**
        * Poll pending transactions for blockchain confirmation
        * Fallback if event subscription doesn't work
        */
       async function pollPendingTransactions() {
         if (!provider) return;
         
         const Transaction = require('../models/Transaction');
         const pending = await Transaction.find({ status: 'pending' });
         
         for (const tx of pending) {
           if (!tx._id) continue;
           
           try {
             // Convert MongoDB ObjectId to bytes32 for matching
             const dbRecordId = ethers.zeroPadValue(tx._id.toBytes32(), 32);
             
             // Query events for this record
             const events = await contract.queryFilter(
               contract.filters.DeviceAction(null, null, null, null, dbRecordId),
               tx.timestamp.getTime() / 1000 - 60, // 1 minute before to 1 minute after
               tx.timestamp.getTime() / 1000 + 60
             );
             
             if (events.length > 0) {
               const event = events[0];
               await confirmTransaction(tx._id.toString(), {
                 txHash: event.transactionHash,
                 blockNumber: event.blockNumber
               });
               console.log(`[BLOCKCHAIN] Confirmed: ${tx._id} -> ${event.transactionHash}`);
             }
           } catch (error) {
             console.error(`[BLOCKCHAIN] Error polling ${tx._id}:`, error.message);
           }
         }
       }
       
       /**
        * Check if blockchain is reachable
        */
       async function isBlockchainReachable() {
         if (!provider) return false;
         try {
           await provider.getBlockNumber();
           return true;
         } catch {
           return false;
         }
       }
       
       module.exports = {
         initBlockchainListener,
         startListening,
         pollPendingTransactions,
         isBlockchainReachable
       };
       ```

       ```javascript
       const express = require('express');
       const { z } = require('zod');
       const { requireAuth } = require('../middleware/auth.middleware');
       const Device = require('../models/Device');
       const { 
         getTransactionHistory, 
         getTransactionSummary,
         getTransactionByHash 
       } = require('../services/transactionRecorder');
       
       const ETHERSCAN_BASE = 'https://sepolia.etherscan.io';
       
       function createTransactionRouter() {
         const router = express.Router();
         
         // Query validation schema
         const historyQuerySchema = z.object({
           limit: z.coerce.number().min(1).max(100).default(50),
           skip: z.coerce.number().min(0).default(0),
           sort: z.enum(['asc', 'desc']).default('desc')
         });
         
         // GET /api/transactions/:deviceId/history
         router.get('/:deviceId/history', requireAuth, async (req, res) => {
           try {
             // Validate device ownership
             const device = await Device.findOne({
               deviceId: req.params.deviceId,
               userId: req.userId
             });
             
             if (!device) {
               return res.status(404).json({ error: 'Device not found' });
             }
             
             // Parse and validate query params
             const { limit, skip, sort } = historyQuerySchema.parse(req.query);
             
             // Get history
             const result = await getTransactionHistory(req.params.deviceId, { limit, skip, sort });
             
             // Add Etherscan links
             const transactions = result.transactions.map(tx => ({
               ...tx,
               etherscanUrl: tx.txHash 
                 ? `${ETHERSCAN_BASE}/tx/${tx.txHash}`
                 : null,
               etherscanAddress: tx.walletAddress
                 ? `${ETHERSCAN_BASE}/address/${tx.walletAddress}`
                 : null
             }));
             
             res.json({
               transactions,
               total: result.total,
               limit: result.limit,
               skip: result.skip,
               hasMore: result.hasMore
             });
           } catch (error) {
             console.error('Error fetching transaction history:', error);
             res.status(500).json({ error: 'Failed to fetch transaction history' });
           }
         });
         
         // GET /api/transactions/:deviceId/summary
         router.get('/:deviceId/summary', requireAuth, async (req, res) => {
           try {
             const device = await Device.findOne({
               deviceId: req.params.deviceId,
               userId: req.userId
             });
             
             if (!device) {
               return res.status(404).json({ error: 'Device not found' });
             }
             
             const summary = await getTransactionSummary(req.params.deviceId);
             
             res.json({
               deviceId: req.params.deviceId,
               ...summary
             });
           } catch (error) {
             console.error('Error fetching summary:', error);
             res.status(500).json({ error: 'Failed to fetch summary' });
           }
         });
         
         // GET /api/transactions/:deviceId/:txId
         router.get('/:deviceId/:txId', requireAuth, async (req, res) => {
           try {
             const device = await Device.findOne({
               deviceId: req.params.deviceId,
               userId: req.userId
             });
             
             if (!device) {
               return res.status(404).json({ error: 'Device not found' });
             }
             
             const Transaction = require('../models/Transaction');
             const tx = await Transaction.findOne({
               _id: req.params.txId,
               deviceId: req.params.deviceId
             });
             
             if (!tx) {
               return res.status(404).json({ error: 'Transaction not found' });
             }
             
             res.json({
               ...tx.toObject(),
               etherscanUrl: tx.txHash 
                 ? `${ETHERSCAN_BASE}/tx/${tx.txHash}`
                 : null,
               etherscanBlock: tx.blockNumber
                 ? `${ETHERSCAN_BASE}/block/${tx.blockNumber}`
                 : null,
               etherscanAddress: tx.walletAddress
                 ? `${ETHERSCAN_BASE}/address/${tx.walletAddress}`
                 : null
             });
           } catch (error) {
             console.error('Error fetching transaction:', error);
             res.status(500).json({ error: 'Failed to fetch transaction' });
           }
         });
         
         // GET /api/transactions/wallet/:walletAddress
         router.get('/wallet/:walletAddress', requireAuth, async (req, res) => {
           try {
             const { limit, skip, sort } = historyQuerySchema.parse(req.query);
             
             // Validate that requested wallet belongs to user
             const User = require('../models/User');
             const user = await User.findById(req.userId);
             
             if (!user || user.walletAddress?.toLowerCase() !== req.params.walletAddress.toLowerCase()) {
               return res.status(403).json({ error: 'Cannot view other user\'s transactions' });
             }
             
             const Transaction = require('../models/Transaction');
             const query = { walletAddress: req.params.walletAddress.toLowerCase() };
             
             const [transactions, total] = await Promise.all([
               Transaction.find(query)
                 .sort({ timestamp: sort === 'asc' ? 1 : -1 })
                 .skip(skip)
                 .limit(limit)
                 .lean(),
               Transaction.countDocuments(query)
             ]);
             
             const txs = transactions.map(tx => ({
               ...tx,
               etherscanUrl: tx.txHash 
                 ? `${ETHERSCAN_BASE}/tx/${tx.txHash}`
                 : null
             }));
             
             res.json({
               transactions: txs,
               total,
               limit,
               skip,
               hasMore: skip + limit < total
             });
           } catch (error) {
             console.error('Error fetching wallet transactions:', error);
             res.status(500).json({ error: 'Failed to fetch transactions' });
           }
         });
         
         return router;
       }
       
       module.exports = { createTransactionRouter };
       ```
