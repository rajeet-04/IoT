    1. Create src/utils/hashChain.js:
       - Import { createHash } from 'node:crypto'
       - Export function computeHash(data): return createHash('sha256').update(data).digest('hex')
       - Export function computeTransactionHash(index, timestamp, deviceId, action, prevHash):
         - const input = `${index}|${timestamp}|${deviceId}|${action}|${prevHash}`
         - return computeHash(input)
       - Export async function verifyChain():
         - Import Transaction model
         - const transactions = await Transaction.find().sort({ index: 1 })
         - If transactions.length === 0: return { valid: true, firstInvalidIndex: null }
         - For i from 0 to transactions.length - 1:
           - const tx = transactions[i]
           - const prevHash = i === 0 ? '0'.repeat(64) : transactions[i-1].hash
           - const expectedHash = computeTransactionHash(tx.index, tx.timestamp.toISOString(), tx.deviceId, tx.action, prevHash)
           - If tx.hash !== expectedHash: return { valid: false, firstInvalidIndex: tx.index }
         - Return { valid: true, firstInvalidIndex: null }
    2. Create src/models/Transaction.js:
       - Import mongoose from 'mongoose', { computeTransactionHash } from '../utils/hashChain.js'
       - Define transactionSchema:
         - index: { type: Number, required: true, unique: true }
         - deviceId: { type: String, required: true, index: true }
         - userId: { type: mongoose.Schema.Types.ObjectId, ref: 'User', required: true }
         - action: { type: String, enum: ['on', 'off'], required: true }
         - timestamp: { type: Date, default: Date.now, required: true }
         - prevHash: { type: String, required: true, default: '0'.repeat(64) }
         - hash: { type: String, required: true }
         - duration: { type: Number }
         - createdAt: { type: Date, default: Date.now }
       - Pre-save middleware (async function):
         - If this.isNew:
           - const lastTx = await this.constructor.findOne({ deviceId: this.deviceId }).sort({ index: -1 })
           - this.index = lastTx ? lastTx.index + 1 : 1
           - this.prevHash = lastTx ? lastTx.hash : '0'.repeat(64)
           - this.hash = computeTransactionHash(this.index, this.timestamp.toISOString(), this.deviceId, this.action, this.prevHash)
       - Block updates: transactionSchema.pre(['findOneAndUpdate', 'updateOne', 'updateMany'], function() { throw new Error('Transactions are immutable') })
       - Block deletes: transactionSchema.pre(['deleteOne', 'deleteMany'], function() { throw new Error('Transactions cannot be deleted') })
       - Compound index: transactionSchema.index({ deviceId: 1, timestamp: -1 })
       - Export default mongoose.model('Transaction', transactionSchema)
  </action>
