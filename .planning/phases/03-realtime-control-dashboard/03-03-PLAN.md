    - On open: sets status to 'connected', calls onConnect callback

    - Store state includes: devices (Device[]), wsStatus ('connecting'|'connected'|'disconnected'|'reconnecting'), pendingCommands (Map)
    - Store actions: fetchDevices() calls GET /api/devices via apiFetch, populates devices array
    - Store actions: sendCommand(deviceId, action, params) — generates commandId, adds to pendingCommands, sends via WebSocket
    - Store actions: updateDevice(deviceId, updates) — merges updates into device in store
    - Store actions: renameDevice(deviceId, name) — calls PUT /api/devices/:id, updates local device name
    - WebSocket message handler updates store:
      - "state_update": updates device state in store
      - "command_response": updates pendingCommand status
      - "command_ack": updates pendingCommand with ack status
      - "presence_change": updates device status (online/offline)

    1. Create frontend/src/lib/ws.ts:
       - Export interface WebSocketClient { connect(): void; disconnect(): void; send(msg: object): boolean; onMessage(cb: (msg: any) => void): void; status: string }
       - Export function createWebSocketClient(backendUrl: string, callbacks?: { onConnect?: () => void; onDisconnect?: () => void }): WebSocketClient
       - Internal state: ws: WebSocket | null, reconnectDelay: number = 1000, maxDelay: number = 1000, reconnectTimer: ReturnType<typeof setTimeout> | null
       - connect():
         - ws = new WebSocket(`${backendUrl}/ws`) — browser sends cookies automatically
         - ws.onopen: reconnectDelay = 1000, callbacks.onConnect?.()
         - ws.onclose: callbacks.onDisconnect?.(), scheduleReconnect()
         - ws.onerror: console.error('WS error'), do not close (let onclose handle reconnect)
         - ws.onmessage: parse JSON, pass to message callbacks
       - scheduleReconnect():
         - reconnectTimer = setTimeout(() => connect(), reconnectDelay)
         - reconnectDelay = Math.min(reconnectDelay * 2, maxDelay)
       - disconnect():
         - clearTimeout(reconnectTimer), reconnectTimer = null
         - if ws: ws.close(), ws = null
       - send(msg): if ws && ws.readyState === WebSocket.OPEN: ws.send(JSON.stringify(msg)); return true; else return false
       - onMessage(cb): store callback, invoke in ws.onmessage handler
    2. Create frontend/src/store/deviceStore.ts:
       - Import { create } from 'zustand', import { apiGet, apiPut } from '../lib/api', import { createWebSocketClient } from '../lib/ws'
       - Define Device type: { id: string; deviceId: string; name: string; status: 'online' | 'offline'; lastSeen: string | null; isConnected: boolean; relayState: boolean }
       - Define PendingCommand type: { deviceId: string; action: string; status: 'pending' | 'success' | 'error'; error?: string }
       - Export const useDeviceStore = create<...>((set, get) => ({
         - devices: [] as Device[],
         - wsStatus: 'disconnected' as string,
         - pendingCommands: new Map<string, PendingCommand>(),
         - wsClient: null as any,
         - init(backendUrl: string):
           - Create wsClient = createWebSocketClient(backendUrl, { onConnect: () => { set({ wsStatus: 'connected' }); const { devices } = get(); devices.forEach(d => wsClient.send({ type: 'subscribe_device', deviceId: d.id })); }, onDisconnect: () => set({ wsStatus: 'reconnecting' }) })
           - wsClient.onMessage((msg) => { handleWsMessage(msg, set, get) })
           - wsClient.connect()
           - set({ wsClient, wsStatus: 'connecting' })
           - get().fetchDevices()
         - async fetchDevices():
           - const res = await apiGet('/api/devices'); const data = await res.json()
           - set({ devices: data.devices.map((d: any) => ({ ...d, relayState: d.status === 'online' })) })
           - After fetch, subscribe to each device via wsClient
         - sendCommand(deviceId: string, action: string, params: object):
           - const commandId = crypto.randomUUID()
           - const { wsClient } = get()
           - set(state => { const newPending = new Map(state.pendingCommands); newPending.set(commandId, { deviceId, action, status: 'pending' }); return { pendingCommands: newPending } })
           - wsClient.send({ type: 'control_command', deviceId, action, params, commandId })
         - async renameDevice(deviceId: string, name: string):
           - await apiPut(`/api/devices/${deviceId}`, { name })
           - set(state => ({ devices: state.devices.map(d => d.id === deviceId ? { ...d, name } : d) }))
         - updateDevice(deviceId: string, updates: Partial<Device>):
           - set(state => ({ devices: state.devices.map(d => d.id === deviceId ? { ...d, ...updates } : d) }))
       }))
       - handleWsMessage(msg, set, get):
         - switch msg.type:
           - "state_update": get().updateDevice(msg.deviceId, { relayState: msg.state })
           - "command_response": set(state => { const newPending = new Map(state.pendingCommands); const cmd = newPending.get(msg.commandId); if (cmd) { newPending.set(msg.commandId, { ...cmd, status: msg.success ? 'success' : 'error', error: msg.error }); } return { pendingCommands: newPending } })
           - "command_ack": (optional — already handled by command_response)
           - "presence_change": get().updateDevice(msg.deviceId, { status: msg.status, isConnected: msg.status === 'online' })
